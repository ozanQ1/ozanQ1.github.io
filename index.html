import tkinter as tk from tkinter import messagebox, ttk import chess import random import time import threading class ChessGame: def __init__(self, root): self.root = root self.root.title("Junie Satranç - Profesyonel Sürüm") self.board = chess.Board() # Tema Tanımları (Lichess ve Modern renk paletleri) self.themes = { "Wood": {"light": "#F0D9B5", "dark": "#B58863", "highlight": "#CDD26A", "move": "#829769"}, "Cello": {"light": "#DEE3E6", "dark": "#8CA2AD", "highlight": "#F6F669", "move": "#BBCB2B"}, "Klasik": {"light": "#DDBB88", "dark": "#AA8844", "highlight": "#FFFF00", "move": "#77DD77"}, "Karanlık": {"light": "#777777", "dark": "#333333", "highlight": "#BBBB00", "move": "#555500"} } self.current_theme = tk.StringVar(value="Wood") # Bot Zorluk Seviyeleri (Derinlik) self.bot_difficulties = { "Başlangıç": 1, "Amatör": 2, "Usta": 3, "Şampiyon": 4 } self.current_difficulty = tk.StringVar(value="Amatör") # Satranç Saati Ayarları (Saniye cinsinden) self.time_white = 600 self.time_black = 600 self.timer_running = False self.square_size = 70 self.selected_square = None self.valid_moves = [] self.move_history = [] # PGN için self.unicode_pieces = { 'R': '♖', 'N': '♘', 'B': '♗', 'Q': '♕', 'K': '♔', 'P': '♙', 'r': '♜', 'n': '♞', 'b': '♝', 'q': '♛', 'k': '♚', 'p': '♟' } self.setup_ui() self.draw_board() self.start_timer() def setup_ui(self): # Ana Frame self.main_frame = tk.Frame(self.root, bg="#2c3e50") self.main_frame.pack(fill="both", expand=True) # Sol Taraf: Tahta self.board_frame = tk.Frame(self.main_frame, bg="#2c3e50") self.board_frame.pack(side="left", padx=20, pady=20) self.canvas = tk.Canvas(self.board_frame, width=self.square_size * 8, height=self.square_size * 8, highlightthickness=0, borderwidth=0) self.canvas.pack() self.canvas.bind("", self.on_square_click) # Sağ Taraf: Panel self.panel = tk.Frame(self.main_frame, bg="#ecf0f1", width=300) self.panel.pack(side="right", fill="y", padx=(0, 20), pady=20) self.panel.pack_propagate(False) tk.Label(self.panel, text="JUNIE CHESS PRO", font=("Segoe UI", 18, "bold"), bg="#ecf0f1", fg="#2c3e50").pack(pady=10) # Saatler self.timer_frame = tk.Frame(self.panel, bg="#ecf0f1") self.timer_frame.pack(fill="x", pady=10) self.label_black_clock = tk.Label(self.timer_frame, text="Siyah: 10:00", font=("Consolas", 14), bg="#34495e", fg="white") self.label_black_clock.pack(fill="x", padx=10, pady=2) self.label_white_clock = tk.Label(self.timer_frame, text="Beyaz: 10:00", font=("Consolas", 14), bg="#ecf0f1", fg="#2c3e50") self.label_white_clock.pack(fill="x", padx=10, pady=2) # Ayarlar settings_frame = tk.Frame(self.panel, bg="#ecf0f1") settings_frame.pack(fill="x", padx=10, pady=10) tk.Label(settings_frame, text="Tema:", bg="#ecf0f1").grid(row=0, column=0, sticky="w") theme_menu = ttk.OptionMenu(settings_frame, self.current_theme, "Wood", *self.themes.keys(), command=lambda _: self.draw_board()) theme_menu.grid(row=0, column=1, sticky="ew", pady=2) tk.Label(settings_frame, text="Zorluk:", bg="#ecf0f1").grid(row=1, column=0, sticky="w") diff_menu = ttk.OptionMenu(settings_frame, self.current_difficulty, "Amatör", *self.bot_difficulties.keys()) diff_menu.grid(row=1, column=1, sticky="ew", pady=2) self.ai_enabled = tk.BooleanVar(value=True) tk.Checkbutton(self.panel, text="Yapay Zekaya Karşı", variable=self.ai_enabled, bg="#ecf0f1").pack(pady=5) # Butonlar btn_frame = tk.Frame(self.panel, bg="#ecf0f1") btn_frame.pack(fill="x", padx=10, pady=5) tk.Button(btn_frame, text="Yeni Oyun", command=self.reset_game, bg="#27ae60", fg="white", relief="flat").pack(side="left", expand=True, fill="x", padx=2) tk.Button(btn_frame, text="Geri Al", command=self.undo_move, bg="#e67e22", fg="white", relief="flat").pack(side="left", expand=True, fill="x", padx=2) # Hamle Geçmişi tk.Label(self.panel, text="Hamle Geçmişi", font=("Segoe UI", 10, "bold"), bg="#ecf0f1").pack(pady=(10,0)) self.history_text = tk.Text(self.panel, height=10, width=30, font=("Consolas", 9)) self.history_text.pack(padx=10, pady=5) self.status_label = tk.Label(self.panel, text="Sıra: Beyaz", font=("Segoe UI", 12), bg="#ecf0f1") self.status_label.pack(pady=10) def draw_board(self): self.canvas.delete("all") theme = self.themes[self.current_theme.get()] # Saat vurguları if self.board.turn == chess.WHITE: self.label_white_clock.config(bg="#f1c40f") self.label_black_clock.config(bg="#34495e") else: self.label_white_clock.config(bg="#ecf0f1") self.label_black_clock.config(bg="#f1c40f") for r in range(8): for c in range(8): x1, y1 = c * self.square_size, r * self.square_size x2, y2 = x1 + self.square_size, y1 + self.square_size color = theme["light"] if (r + c) % 2 == 0 else theme["dark"] square = chess.square(c, 7 - r) if self.selected_square == square: color = theme["highlight"] self.canvas.create_rectangle(x1, y1, x2, y2, fill=color, outline="") # Geçerli hamle işaretçileri (küçük noktalar) if square in self.valid_moves: cx, cy = x1 + self.square_size/2, y1 + self.square_size/2 self.canvas.create_oval(cx-8, cy-8, cx+8, cy+8, fill=theme["move"], outline="", stipple="gray50") # Koordinatlar if c == 0: self.canvas.create_text(x1+8, y1+10, text=str(8-r), font=("Arial", 8, "bold"), fill=theme["dark"] if (r+c)%2==0 else theme["light"]) if r == 7: self.canvas.create_text(x2-8, y2-10, text=chr(97+c), font=("Arial", 8, "bold"), fill=theme["dark"] if (r+c)%2==0 else theme["light"]) piece = self.board.piece_at(square) if piece: unicode_char = self.unicode_pieces.get(piece.symbol()) self.canvas.create_text(x1 + self.square_size/2, y1 + self.square_size/2, text=unicode_char, font=("Arial", 42)) def on_square_click(self, event): col = event.x // self.square_size row = 7 - (event.y // self.square_size) if not (0 <= col < 8 and 0 <= row < 8): return square = chess.square(col, row) if self.selected_square is None: piece = self.board.piece_at(square) if piece and piece.color == self.board.turn: self.selected_square = square self.valid_moves = [m.to_square for m in self.board.legal_moves if m.from_square == square] else: move = chess.Move(self.selected_square, square) promotion_move = chess.Move(self.selected_square, square, chess.QUEEN) final_move = None if move in self.board.legal_moves: final_move = move elif promotion_move in self.board.legal_moves: final_move = promotion_move if final_move: self.execute_move(final_move) if not self.check_game_over() and self.ai_enabled.get() and self.board.turn == chess.BLACK: self.root.after(100, self.make_ai_move) else: # Başka bir taş seçme piece = self.board.piece_at(square) if piece and piece.color == self.board.turn: self.selected_square = square self.valid_moves = [m.to_square for m in self.board.legal_moves if m.from_square == square] else: self.selected_square = None self.valid_moves = [] self.draw_board() def execute_move(self, move): san = self.board.san(move) self.board.push(move) self.move_history.append(san) self.update_history_display() self.selected_square = None self.valid_moves = [] self.draw_board() self.timer_running = True def undo_move(self): if len(self.board.move_stack) > 0: self.board.pop() if self.ai_enabled.get() and len(self.board.move_stack) > 0: self.board.pop() self.move_history = [self.board.san(m) for m in self.board.move_stack] # Basitleştirilmiş tarihçe güncelleme self.update_history_display() self.draw_board() def update_history_display(self): self.history_text.delete("1.0", tk.END) history_str = "" for i in range(0, len(self.move_history), 2): move_num = i // 2 + 1 white_move = self.move_history[i] black_move = self.move_history[i+1] if i+1 < len(self.move_history) else "" history_str += f"{move_num}. {white_move} {black_move}\n" self.history_text.insert(tk.END, history_str) self.history_text.see(tk.END) def start_timer(self): def update_clock(): while True: if self.timer_running and not self.board.is_game_over(): if self.board.turn == chess.WHITE: self.time_white -= 1 if self.time_white <= 0: break else: self.time_black -= 1 if self.time_black <= 0: break self.update_clock_labels() time.sleep(1) threading.Thread(target=update_clock, daemon=True).start() def update_clock_labels(self): def format_time(s): return f"{s//60:02d}:{s%60:02d}" self.label_white_clock.config(text=f"Beyaz: {format_time(self.time_white)}") self.label_black_clock.config(text=f"Siyah: {format_time(self.time_black)}") def make_ai_move(self): depth = self.bot_difficulties[self.current_difficulty.get()] move = self.get_best_move(self.board, depth) if move: self.execute_move(move) self.check_game_over() def get_best_move(self, board, depth): best_move = None best_value = float('inf') if board.turn == chess.BLACK else float('-inf') alpha = float('-inf') beta = float('inf') moves = list(board.legal_moves) moves.sort(key=lambda m: board.is_capture(m), reverse=True) for move in moves: board.push(move) board_value = self.minimax(board, depth - 1, alpha, beta, not (board.turn == chess.WHITE)) board.pop() if board.turn == chess.WHITE: if board_value > best_value: best_value = board_value best_move = move alpha = max(alpha, board_value) else: if board_value < best_value: best_value = board_value best_move = move beta = min(beta, board_value) if beta <= alpha: break return best_move def minimax(self, board, depth, alpha, beta, is_maximizing): if depth == 0 or board.is_game_over(): return self.evaluate_board(board) if is_maximizing: max_eval = float('-inf') for move in board.legal_moves: board.push(move) eval = self.minimax(board, depth - 1, alpha, beta, False) board.pop() max_eval = max(max_eval, eval) alpha = max(alpha, eval) if beta <= alpha: break return max_eval else: min_eval = float('inf') for move in board.legal_moves: board.push(move) eval = self.minimax(board, depth - 1, alpha, beta, True) board.pop() min_eval = min(min_eval, eval) beta = min(beta, eval) if beta <= alpha: break return min_eval def evaluate_board(self, board): if board.is_checkmate(): return -9999 if board.turn == chess.WHITE else 9999 if board.is_game_over(): return 0 piece_values = {chess.PAWN: 10, chess.KNIGHT: 32, chess.BISHOP: 33, chess.ROOK: 50, chess.QUEEN: 90, chess.KING: 900} pst_pawn = [0, 0, 0, 0, 0, 0, 0, 0, 5, 10, 10,-20,-20, 10, 10, 5, 5, -5,-10, 0, 0,-10, -5, 5, 0, 0, 0, 20, 20, 0, 0, 0, 5, 5, 10, 25, 25, 10, 5, 5, 10, 10, 20, 30, 30, 20, 10, 10, 50, 50, 50, 50, 50, 50, 50, 50, 0, 0, 0, 0, 0, 0, 0, 0] score = 0 for square in chess.SQUARES: piece = board.piece_at(square) if piece: val = piece_values.get(piece.piece_type, 0) if piece.piece_type == chess.PAWN: idx = square if piece.color == chess.WHITE else chess.square_mirror(square) val += pst_pawn[idx] * 0.1 score += val if piece.color == chess.WHITE else -val return score def check_game_over(self): if self.board.is_game_over() or self.time_white <= 0 or self.time_black <= 0: self.timer_running = False res = self.board.result() if self.time_white <= 0: res = "0-1 (Zaman)" elif self.time_black <= 0: res = "1-0 (Zaman)" messagebox.showinfo("Oyun Bitti", f"Sonuç: {res}") self.reset_game() return True return False def reset_game(self): self.board = chess.Board() self.selected_square = None self.valid_moves = [] self.move_history = [] self.time_white = 600 self.time_black = 600 self.timer_running = False self.update_history_display() self.update_clock_labels() self.draw_board() if __name__ == "__main__": root = tk.Tk() root.configure(bg="#2c3e50") style = ttk.Style() style.theme_use('clam') game = ChessGame(root) root.mainloop()
